# PRODUCT_VERSION and FILE_NAME needs to be passed to this script

VSVersionInfo(
  ffi=FixedFileInfo(

    # Set not needed items to zero 0.
    filevers=(1, 0, 0, 0),
    prodvers=(1, 0, 0, 0),
    # Contains a bitmask that specifies the valid bits 'flags'r
    mask=0x0,
    # Contains a bitmask that specifies the Boolean attributes of the file.
    flags=0x0,
    # The operating system for which this file was designed.
    # 0x4 - NT and there is no need to change it.
    OS=0x4,
    # The general type of file.
    # 0x1 - the file is an application.
    fileType=0x1,
    # The function of the file.
    # 0x0 - the function is not defined for this fileType
    subtype=0x0,
    # Creation date and time stamp.
    date=(0, 0)
    ),
  kids=[
    StringFileInfo(
      [
      StringTable(
        u'040904e4',
        [StringStruct(u'Comments', u'{name} is an SPV wallet for Bitcoin Cash.'.format(name=PRODUCT_NAME)),
        StringStruct(u'CompanyName', PRODUCT_COMPANY),
        StringStruct(u'FileDescription', u'SPV wallet for Bitcoin Cash'),
        StringStruct(u'FileVersion', PRODUCT_VERSION),
        StringStruct(u'InternalName', FILE_NAME),
        StringStruct(u'LegalCopyright', u'Copyright Â© 2013 - {year} {company} and Electrum Technologies GmbH'.format(year=YEAR, company=PRODUCT_COMPANY)),
        StringStruct(u'LegalTrademarks', u'{name} is a trademark of {company}'.format(name=PRODUCT_NAME, company=PRODUCT_COMPANY)),
        StringStruct(u'OriginalFilename', FILE_NAME),
        StringStruct(u'ProductName', PRODUCT_NAME),
        StringStruct(u'ProductVersion', PRODUCT_VERSION)])
      ]), 
    VarFileInfo([VarStruct(u'Translation', [1033, 1252])])
  ]
)
